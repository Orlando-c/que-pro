var startTime;
var isRunning = false;
onEvent ("startquiz", "click", function() {
  if (!isRunning) {
    isRunning = true;
    startTime = getTime();
    
    timedLoop(1000, function() {
      var currentTime = getTime();
      var elapsedTime = currentTime - startTime;
      elapsedTime = Math.round(elapsedTime/1000);
      setText("timer", "Time: " - elapsedTime + 180);
    });
    
  }
  
});
onEvent("returnhomeend", "click", function() {
  isRunning = false;
  stopTimedLoop();
});
