def letter_to_binary(letter: str) -> str:
    # Convert the letter to ASCII code
    ascii_code = ord(letter)
    
    # Convert the ASCII code to binary
    binary = bin(ascii_code)
    
    # Return the binary representation
    return binary

import math

# Read input from the user
num1 = float(input("Enter a number: "))
num2 = float(input("Enter another number: "))

# Round the numbers up and down
num1_rounded_up = math.ceil(num1)
num1_rounded_down = math.floor(num1)
num2_rounded_up = math.ceil(num2)
num2_rounded_down = math.floor(num2)

# Find the least common multiple of the rounded down numbers
def lcm(a, b):
    return abs(a*b) // math.gcd(a, b)
lcm_rounded_down = lcm(num1_rounded_down, num2_rounded_down)

# Find the greatest common multiple of the rounded up numbers
def gcm(a, b):
    return abs(a*b) // math.gcd(a, b)
gcm_rounded_up = gcm(num1_rounded_up, num2_rounded_up)

# Calculate the factorial of each number
def factorial(n):
    if n == 0:
        return 1
    else:
        return n * factorial(n-1)
num1_factorial = factorial(num1)
num2_factorial = factorial(num2)

# Print the results
print("Rounded up:", num1_rounded_up, num2_rounded_up)
print("Rounded down:", num1_rounded_down, num2_rounded_down)
print("Least common multiple of rounded down numbers:", lcm_rounded_down)
print("Greatest common multiple of rounded up numbers:", gcm_rounded_up)
print("Factorials:", num1_factorial, num2_factorial)

3.17 Algorithmic Efficiency

Vocab

  • Problem: a general description of a task that can or cannot be solved algorithmically
  • Decision Problem: A problem with a yes or no answer
  • Organization Problem: a problem with a goal of finding the best answer
  • Instance: a problem with a specific input
  • Efficiency: amount of computing needed to solve a problem
  • Polynomial Efficiency (Good): more work takes a proportional amount of time (1 job is +2 time)
  • Exponential Efficiency (Bad): more work takes an exponential amount more time (1 job is 2x time)
  • Heuristic Approach: When optimal solutions are inefficient, look for a possibly optimal solution that is more efficient
  • Decidable Problem: A decision problem that has a clear solution that will always make a correct output
  • Undecidable Problem: A decision problem with no solution that is not guaranteed to produce the correct output

from itertools import permutations

def get_travel_time(route, times):
  total_time = 0
  for i in range(len(route) - 1):
    total_time += times[route[i]][route[i+1]]
  return total_time

def find_fastest_route(times):
  locations = list(range(5))
  routes = permutations(locations)
  
  # Find the fastest route
  fastest_time = float('inf')
  fastest_route = None
  for route in routes:
    time = get_travel_time(route, times)
    if time < fastest_time:
      fastest_time = time
      fastest_route = route
      
  return fastest_route
